${getCategoryName(tool.category)}

${tool.name}

${tool.description}

${tool.tags.map(tag => `${tag}`).join('')}

价格：

${tool.price}
查看详情
`).join(''); toolsContainer.innerHTML = toolCards; console.log('工具卡片生成完成，共生成', data.tools.length, '个卡片'); }) .catch(error => { console.error('加载工具数据失败:', error); // 显示错误提示给用户 const toolsContainer = document.getElementById('tools-container'); if (toolsContainer) { toolsContainer.innerHTML = `

加载工具失败: ${error.message}

请检查tools.json文件是否存在或格式是否正确

`; } }); } // 获取分类显示名称 function getCategoryName(category) { const categories = { 'content': 'AI聊天助手', 'image': 'AI图像助手', 'video': 'AI视频工具', 'automation': 'AI办公工具', 'more': '虚拟角色互动' }; return categories[category] || category; } // 页面加载时执行 window.addEventListener('DOMContentLoaded', () => { loadTools(); // ... 保留其他现有事件监听 ... });
